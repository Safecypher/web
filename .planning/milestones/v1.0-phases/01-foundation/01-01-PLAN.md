2. Move existing conflicting files to `_archive/` (do NOT delete — preserve history):
   - Move `index.html` → `_archive/index.html`
   - Move `safe-verify-landing.html` → `_archive/safe-verify-landing.html`
   - Move `assets/` → `_archive/assets/`

3. Remove Astro-only packages from `package.json` before scaffolding. Run:
   ```bash
   npm uninstall astro
   ```

4. Scaffold Next.js in-place (the repo directory already exists, use `.`):
   ```bash
   npx create-next-app@latest . --typescript --eslint --tailwind --app --src-dir --import-alias "@/*" --no-git
   ```
   When prompted about existing files: accept overwrite for package.json, tsconfig.json, .gitignore. Keep existing files for anything not listed.

   IMPORTANT flags:
   - `--src-dir`: puts app/ inside src/ → paths become src/app/, src/components/, etc.
   - `--tailwind`: installs Tailwind but we'll reconfigure for v4 in Plan 01-03
   - `--no-git`: repo already has git, don't reinitialise
   - `--import-alias "@/*"`: sets up the @/ alias pointing to src/

5. After scaffold, verify `src/` contains: `src/app/`, `src/app/layout.tsx`, `src/app/page.tsx`, `src/app/globals.css`

   ```bash
   npm install daisyui@latest
   ```

   import { Outfit, Playfair_Display } from 'next/font/google'
   import './globals.css'

   const outfit = Outfit({
     subsets: ['latin'],
     variable: '--font-sans',
     display: 'swap',
   })

   const playfair = Playfair_Display({
     subsets: ['latin'],
     variable: '--font-serif',
     display: 'swap',
   })

   export const metadata: Metadata = {
     title: 'SafeCypher — Eliminate Card-Not-Present Fraud',
     description: 'Dynamic security codes that eliminate CNP fraud for card issuers.',
   }

   export default function RootLayout({
     children,
   }: {
     children: React.ReactNode
   }) {
     return (
       <html lang="en" data-theme="safecypher-dark" className={`${outfit.variable} ${playfair.variable}`}>
         <body className="antialiased">
           {children}
         </body>
       </html>
     )
   }
